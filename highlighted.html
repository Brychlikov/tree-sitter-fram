5pub let isEmpty xs =
6 match xs with
7 | [] => True
8 | _ => False
9 end

11pub let length xs =
12 let rec lengthAux xs (res : Int) =
13 match xs with
14 | [] => res
15 | x :: xs => lengthAux xs (res + 1)
16 end
17 in lengthAux xs 0

43pub let nth xs (n : Int) =
44 let rec nthAux xs (n : Int) =
45 match xs with
46 | [] => None
47 | x :: xs => if n == 0 then Some x else nthAux xs (n - 1)
48 end
49 in if n < 0 then None else nthAux xs n

51pub let nthErr { ~onError } xs (n : Int) =
52 let rec nthErrAux xs (n : Int) =
53 match xs with
54 | [] => ~onError ()
55 | x :: xs => if n == 0 then x else nthErrAux xs (n - 1)
56 end
57 in if n < 0 then ~onError () else nthErrAux xs n

337pub let foldLeftMap f acc xs =
338 let rec foldLeftMapAux acc l_acc xs =
339 match xs with
340 | [] => acc, rev l_acc
341 | x :: xs =>
342 let (acc, x) = f acc x in
343 foldLeftMapAux acc (x :: l_acc) xs
344 end
345 in foldLeftMapAux acc [] xs
346
347pub let foldLeftFilterMap f acc xs =
348 let rec foldLeftFilterMapAux acc l_acc xs =
349 match xs with
350 | [] => acc, rev l_acc
351 | x :: xs =>
352 match f acc x with
353 | None => foldLeftFilterMapAux acc l_acc xs
354 | Some (acc, x) => foldLeftFilterMapAux acc (x :: l_acc) xs
355 end
356 end
357 in foldLeftFilterMapAux acc [] xs

359pub let foldLeftConcatMap f acc xs =
360 let (acc, xs) = foldLeftMap f acc xs
361 in (acc, concat xs)

365pub let mem
366 { E, method equal : E -> E ->[] Bool }
367 (v : E) (xs : List E) =
368 let rec memAux (xs : List E) =
369 match xs with
370 | [] => False
371 | x :: xs => v == x || memAux xs
372 end
373 in memAux xs

375pub let equal
376 { E, method equal : E -> E ->[] Bool }
377 (xs : List E) (ys : List E) =
378 let rec equalAux (xs : List E) (ys : List E) =
379 match xs, ys with
380 | [], [] => True
381 | x :: xs, y :: ys => x == y && equalAux xs ys
382 | _, _ => False
383 end
384 in equalAux xs ys
